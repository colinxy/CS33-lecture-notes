** Jumping

*** Conditional Branch example

#+BEGIN_SRC C

  long absdiff(long x, long y) {
    long result;
    if (x < y)
      result = x - y;
    else
      result = y - x;
    return result;
  }

#+END_SRC

| register | variable |
|          |          |

#+BEGIN_SRC asm

  absdiff:
        cmpq      %rsi, %rdi
        jle       .L4
        movq      %rdi, %rax
        subq      %rsi, %rax
        ret
  .L4
        movq      %rsi, %rax
        subq      %rdi, %rax
        ret

#+END_SRC
